void create_symlink() {
    const char *source_path = "mouse_data.dat"; // Path to the generated file
    const char *target_path = "../mouse_plotter/build/mouse_data.dat"; // Symlink location

    // Remove existing symbolic link if it exists
    unlink(target_path);

    // Create the symbolic link
    if (symlink(source_path, target_path) == -1) {
        perror("Error creating symbolic link");
    } else {
        printf("Symbolic link created: %s -> %s\n", target_path, source_path);
    }
}
